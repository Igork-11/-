#include <stdio.h>
#include <locale.h>
#include <math.h>

int main()
{
    int a = 0, b = 0, c = 0, i = 1, j = 1, l = 1, k = 0, k1 = 0, k2 = 0;
    do{
        if (i % 2 != 0)
        {
        a += i;
        k+=1;
        }
        i++;
        
    } 
    while (i < 80);
    for(j=1; j < 80; j++)
    {
        if (j % 2 != 0)
        {
        b += j;
        k1+=1;
        }
        
    }
    while(l<80)
    {
        if (l % 2 != 0)
        {
        c += l;
        k2+=1;
        }
        l++;
    }
    a=a/k;
    b=b/k1;
    c=c/k2;
    printf("a=%d \n",a);
    printf("b=%d \n",b);
    printf("c=%d \n",c);
    return 0;
}

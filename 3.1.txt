#include <stdio.h>
#include <math.h>
int main()
{
float a,b,c,y;
printf("a=");
scanf("%g",&a);
printf("b=");
scanf("%g",&b);
printf("c=");
scanf("%g",&c);
if(c>0){
y=sqrt(b-a)+2*sqrt(c);
printf("y="); 
printf("%g", y); 
}
else if(c<0)
{
y=b+12*pow(c,3);
    printf("y="); 
    printf("%g", y); 
}
else
{
y=b/(c-6*a);
    printf("y="); 
    printf("%g", y); 
}
    return 0;
}

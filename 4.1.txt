#include <stdio.h>
#include <locale.h>
#include <math.h>

int main()
{
    float x,y,z,t,n,m,i;
    do
    {
    printf("x=");
    scanf("%f",&x);
    printf("y=");
    scanf("%f",&y);
    printf("z=");
    scanf("%f",&z);
    n=x+(y/z);
    m=x-(1/(1+pow(x,2)));
    if(y==0 || z==0 || n==0 || m==0)
    {
        printf("Помилка\n");
    }
    else
    {
    t=(1+z)*(n/m);
    printf("%f",t);
    }
    }while(y==0 || z==0 || n==0 || m==0);
    
    return 0;
}

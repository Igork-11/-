#include <stdio.h>
#include <math.h>

int main()
{
    float a,y,z,cos2,b,x;
    printf("a=");
    scanf("%f",&a);
    printf("y=");
    scanf("%f",&y);
    printf("z=");
    scanf("%f",&z);
    b=pow(y,2)-2*a;
    if(b>0){
        cos2=(1+cos(2*z))/2;
        x=a*b+cos2;
        printf("%f",x); 
    }
    else
    {
       printf("Oшибка"); 
    }
    return 0;
}
